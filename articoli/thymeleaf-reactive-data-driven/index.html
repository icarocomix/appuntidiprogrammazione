Esigenza Reale

Visualizzare i primi risultati di una ricerca massiva mentre il server sta ancora elaborando il resto della query.

Analisi Tecnica

Problema: Latenza percepita alta (schermata bianca) durante l’elaborazione di grandi volumi di dati lato server.

Perché: Uso il rendering reattivo. Ho scelto di inviare i dati in streaming per permettere al browser di iniziare il parsing del DOM prima che la risposta HTTP sia completata.

/* Controller WebFlux con rendering data-driven: il template viene inviato a
    chunk man mano che i dati arrivano dal database reattivo. */
@Controller public class InventoryController {
    @Autowired private ProductRepository productRepository;
    // Repository R2DBC reattivo @GetMapping("/inventory") public String
        inventory(Model model)
    {
        // Flux<Product>: i dati arrivano dal DB in modo asincrono Flux<Product>
            products = productRepository.findAllByOrderByNameAsc()
            .map(this::enrichProduct) // Arricchimento asincrono
            .onErrorResume(e ->
        {
            log.error("Errore nel recupero prodotti", e);
            return Flux.empty();
        }
        );
        // ReactiveDataDriverContextVariable: // - Primo parametro: il Flux con
            i dati // - Secondo parametro: chunk size (10 = invia 10 righe HTML
            alla volta al browser) model.addAttribute("products", new
            ReactiveDataDriverContextVariable(products, 10))
        ;
        // I dati non-reattivi vengono aggiunti normalmente
            model.addAttribute("pageTitle", "Inventario Prodotti")
        ;
        model.addAttribute("totalCount", productRepository.count());
        // Mono<Long>: gestito automaticamente
        return "inventory";
    }
    @GetMapping("/search") public String search(
    @RequestParam String query, Model model) {
        // Combino stream reattivi: ricerca full-text + dati di stock in
            parallelo Flux<ProductSearchResult> results = productRepository
            .findByNameContainingIgnoreCase(query)
            .zipWith(stockService.getStockLevels()) // Join reattivo .map(tuple
            -> new ProductSearchResult(tuple.getT1(), tuple.getT2()))
        ;
        // Chunk di 5: i risultati appaiono nel browser appena sono disponibili
            model.addAttribute("results", new
            ReactiveDataDriverContextVariable(results, 5))
        ;
        model.addAttribute("query", query);
        return "search-results";
    }
}

// </tr> // </tbody> // </table> // </body> // </html> /* Configurazione WebFlux
    per ottimizzare lo streaming: */ @Configuration public class WebFluxConfig
    implements WebFluxConfigurer
{
    @Override public void configureHttpMessageCodecs(ServerCodecConfigurer
        configurer) {
        configurer.defaultCodecs().maxInMemorySize(16 * 1024 * 1024);
        // 16MB buffer
    }
    @Bean public ThymeleafReactiveViewResolver thymeleafReactiveViewResolver(
        SpringWebFluxTemplateEngine engine) {
        ThymeleafReactiveViewResolver resolver = new
            ThymeleafReactiveViewResolver();
        resolver.setTemplateEngine(engine);
        resolver.setResponseMaxChunkSizeBytes(8192);
        // 8KB per chunk HTTP
        return resolver;
    }
}
