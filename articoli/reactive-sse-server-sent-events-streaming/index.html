Esigenza Reale

Creare una dashboard che si aggiorna automaticamente man mano che arrivano nuovi dati dal database o da un message broker.

Analisi Tecnica

Problema: Necessità di aggiornamenti real-time che solitamente richiederebbero polling pesante o l’overhead di un server WebSocket.

Perché: Uso il Data-Driven Streaming. Ho scelto SSE con Thymeleaf Reactive per inviare chunk di HTML pronti per il DOM non appena il backend produce un nuovo evento, riducendo la latenza percepita.

/* Controller WebFlux che espone due endpoint: 1) La pagina HTML iniziale con il
    container vuoto 2) Lo stream SSE che invia frammenti HTML al client */
@Controller
@RequestMapping("/monitoring") public class MonitoringController {
    @Autowired private MetricsService metricsService;
    @Autowired private TemplateEngine templateEngine;
    /* Endpoint per la pagina iniziale: renderizza il layout con il container
        vuoto. */
    @GetMapping public String dashboardPage(Model model) {
        // Dati iniziali sincroni: gli aggiornamenti arrivano via SSE
            model.addAttribute("title", "Monitoring Dashboard")
        ;
        return "monitoring/dashboard";
    }
    /* Endpoint SSE: restituisce frammenti HTML renderizzati lato server. Il
        client (HTMX o EventSource) riceve HTML pronto da inserire nel DOM. */
    @GetMapping(value = "/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
        public Flux<ServerSentEvent<String>> streamMetrics() {
        return metricsService.getMetricsStream()
        // Flux<MetricEvent> da Kafka/Redis/DB .map(metric ->
        {
            // Renderizzo il frammento Thymeleaf per ogni evento Context ctx =
                new Context()
            ;
            ctx.setVariable("metric", metric);
            ctx.setVariable("timestamp", LocalDateTime.now());
            String html = templateEngine.process(
                "monitoring/fragments/metric-row",
            // Frammento specifico ctx )
            ;
            return ServerSentEvent.<String>builder()
                .id(metric.getId().toString()) .event("metric-update")
            // Tipo evento: il client filtra per tipo .data(html) // HTML
                pre-renderizzato .build()
            ;
        }
        ) .onErrorResume(e -> {
            log.error("Errore nello stream metriche", e);
            return Flux.just(ServerSentEvent.<String>builder() .event("error")
                .data("<div class='alert alert-danger'>Connessione persa</div>")
                .build());
        }
        );
    }
    /* Per stream con backpressure: limito la frequenza di emissione per non
        sovraccaricare il client. */
    @GetMapping(value = "/logs", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
        public Flux<ServerSentEvent<String>> streamLogs() {
        return logService.getLogStream() .sample(Duration.ofMillis(500))
        // Massimo 2 aggiornamenti al secondo .window(10) // Raggruppo in batch
            da 10 log .flatMap(batch -> batch.collectList()) .map(logs ->
        {
            Context ctx = new Context();
            ctx.setVariable("logs", logs);
            String html =
                templateEngine.process("monitoring/fragments/log-batch", ctx);
            return ServerSentEvent.<String>builder() .event("log-batch")
                .data(html) .build();
        }
        );
    }
}
