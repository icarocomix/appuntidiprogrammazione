/* In Spring Boot, schedulo un job che verifica quotidianamente 
    l'integrità dell'archivio e la freschezza dell'ultimo backup:
@Scheduled(cron = "0 0 6 * * *")
// Ogni mattina alle 06:00
public void verifyBackupHealth() {
    try {
        Process p = Runtime.getRuntime().exec("pgbackrest 
            --stanza=myapp check");
        int exit = p.waitFor();
        if (exit != 0) {
            alertService.sendCritical("pgBackRest check fallito: 
                backup PITR non affidabile");
        }
    } catch (Exception e) {
        alertService.sendCritical("Impossibile verificare il 
            backup: " + e.getMessage());
    }
}
