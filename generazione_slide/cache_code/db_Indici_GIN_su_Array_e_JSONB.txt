    (attributes);

CREATE INDEX idx_articles_tags ON articles USING GIN (tags);
-- Trova articoli che contengono TUTTI i tag specificati:
SELECT id, title FROM articles WHERE tags @> ARRAY['postgresql', 
    'performance'];

SELECT id, title FROM articles WHERE tags && ARRAY['postgresql', 
    'indexing'];
