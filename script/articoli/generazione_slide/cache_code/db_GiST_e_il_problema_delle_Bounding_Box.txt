CREATE INDEX idx_geo_polygons ON city_zones USING GIST (geom);
CLUSTER city_zones USING idx_geo_polygons;
ANALYZE city_zones;
EXPLAIN (ANALYZE, BUFFERS)
SELECT zone_name, municipality FROM city_zones
WHERE ST_Contains(geom, ST_SetSRID(ST_MakePoint(10.2035, 
    45.5416), 4326));
-- Confronto "Buffers: shared read" prima e dopo il CLUSTER: se 
    le bounding box si sovrappongono meno, il Planner percorre 
    meno rami e legge meno pagine disco.
SELECT zone_name, geom <-> ST_SetSRID(ST_MakePoint(10.2035, 
    45.5416), 4326) AS dist
FROM city_zones
ORDER BY dist LIMIT 5;
